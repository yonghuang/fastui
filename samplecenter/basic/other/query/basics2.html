  $("#TxtMind99").val($("select[name=agree] option:selected").text());
  等同于
  fastDev("#TxtMid99").prop("value", fastDev("select[name=agree] option:selected").getText());
		
  $("#writeMind").hide();
  等同于
  fastDev("#writeMind").hide();
	
  $("#pro.name").show();
  等同于
  fastDev("[id='pro.name']").show();
		
  $("select[name=agree]").val("true");
  等同于
  fastDev("select[name=agree]").prop("value", "true")
		
  $("#driverFee").attr("disabled","true");
  等同于
  fastDev("#driverFee").attr("disabled", "true")
		
  $(function(){
  等同于
  fastDev(function() {
 	//getFirstActName();
  });
		
  $("#writeMind").is(":hidden")==false
  等同于
  fastDev("#writeMind:hidden").elems.length > 0
